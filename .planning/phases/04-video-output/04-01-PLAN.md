  <action>
Create Promise-based FFmpeg spawn wrapper following research patterns.

```typescript
// src/encoder/ffmpeg-process.ts
import { spawn, ChildProcess } from 'node:child_process';
import { EventEmitter } from 'node:events';
import ffmpegPath from 'ffmpeg-static';

export interface FFmpegProcessOptions {
  args: string[];
  stdin?: 'pipe' | 'ignore';
  onStderr?: (data: string) => void;
}

export interface FFmpegProcess {
  process: ChildProcess;
  stdin: NodeJS.WritableStream | null;
  finished: Promise<void>;
}

export function spawnFFmpeg(options: FFmpegProcessOptions): FFmpegProcess {
  const { args, stdin = 'ignore', onStderr } = options;

  if (!ffmpegPath) {
    throw new Error('FFmpeg binary not found. Ensure ffmpeg-static is installed.');
  }

  const ffmpeg = spawn(ffmpegPath, [
    '-hide_banner',
    '-loglevel', 'error',
    ...args
  ], {
    stdio: [stdin, 'pipe', 'pipe']
  });

  if (onStderr) {
    ffmpeg.stderr?.on('data', (data: Buffer) => {
      onStderr(data.toString());
    });
  }

  const finished = new Promise<void>((resolve, reject) => {
    ffmpeg.on('close', (code) => {
      if (code === 0) resolve();
      else reject(new Error(`FFmpeg exited with code ${code}`));
    });
    ffmpeg.on('error', reject);
  });

  return {
    process: ffmpeg,
    stdin: stdin === 'pipe' ? ffmpeg.stdin : null,
    finished
  };
}
```

Key decisions from research:
- Always include `-hide_banner` and `-loglevel error` to reduce noise
- Input options MUST come before -i flag (enforced by caller building args array)
- Return Promise-based interface for async/await usage
- Expose stdin for frame piping (VideoEncoder will use this)

// src/schemas/audio.ts
import { z } from 'zod';

/**
 * Audio configuration for video encoding
 * Supports background audio with volume control and fade effects
 */
export const AudioConfigSchema = z.object({
  /** Audio file path or URL */
  src: z.string().min(1, 'Audio source path is required'),

  /** Volume multiplier 0.0-1.0 (default: 1.0) - AUDI-02 */
  volume: z.number().min(0).max(1).default(1),

  /** Fade in duration in seconds (default: 0) - AUDI-03 */
  fadeIn: z.number().min(0).default(0),

  /** Fade out duration in seconds (default: 0) - AUDI-04 */
  fadeOut: z.number().min(0).default(0),
}).strict();

export type AudioConfig = z.infer<typeof AudioConfigSchema>;
```

Update src/schemas/index.ts to export the new schema:

```

Schema follows existing patterns:
- Use `.strict()` to reject unknown fields (matches other schemas)
- Sensible defaults (volume=1, fades=0)
- Min validation for seconds (no negative durations)
