```typescript
  open: boolean;
  onOpenChange: (open: boolean) => void;
  status: 'queued' | 'processing' | 'completed' | 'failed' | null;
  progress?: number;
  error?: string;
}
```

Imports:
```typescript
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from '@/components/ui/dialog';
import { Progress } from '@/components/ui/progress';
import { CheckCircle, XCircle, Loader2 } from 'lucide-react';
```

Behavior:
- Dialog cannot be closed while status is 'queued' or 'processing'
- onOpenChange only fires if canClose = (status === 'completed' || status === 'failed')

UI States:

**queued:**
- Title: "Preparing render..."
- Loader2 icon spinning
- Text: "Waiting in queue..."

**processing:**
- Title: "Rendering video..."
- Progress bar (value = progress || 0)
- Text: "{progress || 0}% complete"

**completed:**
- Title: "Render complete!"
- CheckCircle icon (green-400)
- Text: "Video rendered successfully!"
- Subtext: "Opening in your video player..."

**failed:**
- Title: "Render failed"
- XCircle icon (red-400)
- Text: "Render failed"
- Error message displayed if error prop provided

Use dark theme colors:
- Dialog content: dark background (default from shadcn)
- Icons: text-green-400 for success, text-red-400 for error
- Progress text: text-slate-400

Add state and hook:
```typescript
const [progressDialogOpen, setProgressDialogOpen] = useState(false);
const { render, isRendering, progress, resetProgress } = useRenderVideo({
  onComplete: async (videoId) => {
    // Auto-open video in system player
    try {
      await fetch(`/studio/videos/${videoId}/open`, { method: 'POST' });
    } catch (error) {
      console.error('Failed to auto-open video:', error);
    }
  },
});
```

Add "Render Video" button in the dialog footer (next to Edit button):
```typescript
<Button
  onClick={() => {
    if (template) {
      setProgressDialogOpen(true);
      render(template.id);
    }
  }}
  disabled={isRendering}
  className="gap-2"
>
  <Play className="size-4" />
  Render Video
</Button>
```

Add RenderProgressDialog at the end of the component:
```typescript
<RenderProgressDialog
  open={progressDialogOpen}
  onOpenChange={(open) => {
    setProgressDialogOpen(open);
    if (!open) {
      resetProgress();
    }
  }}
  status={progress?.status || null}
  progress={progress?.progress}
  error={progress?.error}
/>
```
