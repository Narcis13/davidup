import type { Canvas, SKRSContext2D } from '@napi-rs/canvas';
import { createCanvas } from '@napi-rs/canvas';
import type { Element } from '../schemas/elements.js';
import type { Scene } from '../schemas/scene.js';
import { RendererRegistry } from './renderer-registry.js';
import { AssetManager } from './asset-manager.js';
import { applyTransforms } from './transforms.js';
import { Timeline, SceneInfo, renderTransition } from '../timeline/index.js';
import {
  getAnimatedValue,
  generateEnterKeyframes,
  generateExitKeyframes,
  type PropertyAnimation,
  type Keyframe,
  type EasingName,
} from '../animation/index.js';

export interface AnimatedElement extends Element {
  /** Property animations */
  animations?: PropertyAnimation[];
  /** Enter animation preset */
  enter?: {
    type: 'fade' | 'slide' | 'scale' | 'bounce';
    duration: number;
    direction?: 'left' | 'right' | 'top' | 'bottom';
    distance?: number;
    easing?: EasingName;
  };
  /** Exit animation preset */
  exit?: {
    type: 'fade' | 'slide' | 'scale' | 'bounce';
    duration: number;
    direction?: 'left' | 'right' | 'top' | 'bottom';
    distance?: number;
    easing?: EasingName;
  };
  /** When element becomes visible (seconds from scene start) */
  startTime?: number;
  /** When element becomes invisible (seconds from scene start) */
  endTime?: number;
}

export interface AnimatedScene extends Scene {
  elements: AnimatedElement[];
}

export interface AnimatedFrameGeneratorConfig {
  width: number;
  height: number;
  fps: number;
  scenes: AnimatedScene[];
}

/**
 * Frame generator with animation support.
 *
 * Extends the basic frame generator to:
 * - Process keyframe animations on elements
 * - Apply enter/exit animation presets
 * - Handle multi-scene timelines
 * - Render scene transitions
 */
export class AnimatedFrameGenerator {
  private readonly canvas: Canvas;
  private readonly ctx: SKRSContext2D;
  private readonly config: AnimatedFrameGeneratorConfig;
  private readonly registry: RendererRegistry;
  private readonly assets: AssetManager;
  private readonly timeline: Timeline;

  constructor(
    config: AnimatedFrameGeneratorConfig,
    registry: RendererRegistry,
    assets: AssetManager
  ) {
    this.config = config;
    this.registry = registry;
    this.assets = assets;
    this.canvas = createCanvas(config.width, config.height);
    this.ctx = this.canvas.getContext('2d');
    this.timeline = new Timeline({
      scenes: config.scenes,
      fps: config.fps,
    });
  }

  /**
   * Get total number of frames in the video.
   */
  getTotalFrames(): number {
    return this.timeline.getTotalFrames();
  }

  /**
   * Generate a single frame at the specified frame number.
   */
  generateFrame(frameNumber: number): Buffer {
    const sceneInfo = this.timeline.getSceneAtFrame(frameNumber);

    if (sceneInfo.inTransition && sceneInfo.transition) {
      return this.renderTransitionFrame(sceneInfo);
    }

    return this.renderSceneFrame(sceneInfo);
  }

  /**
   * Generate all frames for the video.
   *
   * @returns Generator yielding frame buffers
   */
  *generateAllFrames(): Generator<Buffer, void, unknown> {
    const totalFrames = this.getTotalFrames();
    for (let frame = 0; frame < totalFrames; frame++) {
      yield this.generateFrame(frame);
    }
  }

  /**
   * Render a normal scene frame (no transition).
   */
  private renderSceneFrame(sceneInfo: SceneInfo): Buffer {
    const { scene, localFrame } = sceneInfo;

    // Clear and fill background
    this.ctx.fillStyle = scene.background ?? '#000000';
    this.ctx.fillRect(0, 0, this.config.width, this.config.height);

    // Render each visible element with animations applied
    for (const element of (scene.elements as AnimatedElement[])) {
      const animatedElement = this.resolveAnimatedElement(
        element,
        localFrame,
        this.config.fps
      );

      if (animatedElement) {
        this.renderElement(animatedElement);
      }
    }

    return Buffer.from(this.canvas.data());
  }

  /**
   * Render a transition frame between two scenes.
   */
  private renderTransitionFrame(sceneInfo: SceneInfo): Buffer {
    const { transition, localFrame } = sceneInfo;
    if (!transition) {
      return this.renderSceneFrame(sceneInfo);
    }

    // Render "from" scene
    const fromSceneInfo: SceneInfo = {
      ...sceneInfo,
      inTransition: false,
      transition: undefined,
    };
    const fromBuffer = this.renderSceneFrame(fromSceneInfo);

    // Calculate local frame for "to" scene (starts at 0)
    const toSceneInfo: SceneInfo = {
      scene: transition.to,
      sceneIndex: sceneInfo.sceneIndex + 1,
      localFrame: 0,  // "to" scene is at its beginning
      globalFrame: sceneInfo.globalFrame,
      inTransition: false,
    };
    const toBuffer = this.renderSceneFrame(toSceneInfo);

    // Composite with transition
    return renderTransition({
      fromBuffer,
      toBuffer,
      width: this.config.width,
      height: this.config.height,
      progress: transition.easedProgress,
      type: transition.type,
      direction: transition.direction,
    });
  }

  /**
   * Resolve animated properties for an element at a specific frame.
   *
   * @returns The element with animated values, or null if element is not visible
   */
  private resolveAnimatedElement(
    element: AnimatedElement,
    localFrame: number,
    fps: number
  ): Element | null {
    const localTime = localFrame / fps;

    // Check visibility based on startTime/endTime
    const startTime = element.startTime ?? 0;
    const endTime = element.endTime ?? Infinity;

    if (localTime < startTime || localTime > endTime) {
      return null;
    }

    // Calculate frame relative to element's start
    const elementLocalFrame = localFrame - Math.round(startTime * fps);

    // Collect all animations (custom + presets)
    const allAnimations = this.collectAnimations(element, fps);

    // Start with the base element
    const resolved: any = { ...element };

    // Apply each animation
    for (const anim of allAnimations) {
      if (anim.keyframes.length > 0) {
        const value = getAnimatedValue(elementLocalFrame, anim.keyframes);

        // Handle special clamping for certain properties
        if (anim.property === 'opacity') {
          resolved.opacity = Math.max(0, Math.min(1, value));
        } else if (anim.property === 'scaleX' || anim.property === 'scaleY') {
          // Store scale in transform property
          resolved[anim.property] = Math.max(0, value);
        } else {
          resolved[anim.property] = value;
        }
      }
    }

    return resolved as Element;
  }

  /**
   * Collect all animations for an element, including presets.
   */
  private collectAnimations(
    element: AnimatedElement,
    fps: number
  ): PropertyAnimation[] {
    const animations: PropertyAnimation[] = [];

    // Add custom animations
    if (element.animations) {
      // Convert time-based keyframes to frame-based
      for (const anim of element.animations) {
        animations.push({
          property: anim.property,
          keyframes: anim.keyframes.map(kf => ({
            ...kf,
            frame: Math.round(kf.frame),  // Assuming keyframes are already in frames
          })),
        });
      }
    }

    // Add enter preset animations
    if (element.enter) {
      const enterDurationFrames = Math.round(element.enter.duration * fps);
      const enterAnims = generateEnterKeyframes(
        {
          type: element.enter.type,
          duration: enterDurationFrames,
          direction: element.enter.direction,
          distance: element.enter.distance,
          easing: element.enter.easing,
        },
        0,  // Start at frame 0 of element
        (element as any).x ?? 0,
        (element as any).y ?? 0
      );
      animations.push(...enterAnims);
    }

    // Add exit preset animations
    if (element.exit && element.endTime !== undefined) {
      const exitDurationFrames = Math.round(element.exit.duration * fps);
      const startTime = element.startTime ?? 0;
      const visibleDurationFrames = Math.round((element.endTime - startTime) * fps);
      const exitStartFrame = visibleDurationFrames - exitDurationFrames;

      const exitAnims = generateExitKeyframes(
        {
          type: element.exit.type,
          duration: exitDurationFrames,
          direction: element.exit.direction,
          distance: element.exit.distance,
          easing: element.exit.easing,
        },
        exitStartFrame,
        (element as any).x ?? 0,
        (element as any).y ?? 0
      );
      animations.push(...exitAnims);
    }

    return animations;
  }

  /**
   * Render a single element to the canvas.
   */
  private renderElement(element: Element): void {
    this.ctx.save();

    try {
      // Apply transforms including animated values
      applyTransforms(this.ctx, {
        x: (element as any).x ?? 0,
        y: (element as any).y ?? 0,
        rotation: (element as any).rotation ?? 0,
        scaleX: (element as any).scaleX ?? 1,
        scaleY: (element as any).scaleY ?? 1,
        opacity: (element as any).opacity ?? 1,
      });

      // Render via registry
      this.registry.render(this.ctx, element, this.assets);
    } finally {
      this.ctx.restore();
    }
  }
}

/**
 * Factory function to create a configured animated renderer.
 */
export function createAnimatedRenderer(
  config: Omit<AnimatedFrameGeneratorConfig, 'fps' | 'scenes'> & {
    fps?: number;
    scenes?: AnimatedScene[];
  }
): {
  generator: AnimatedFrameGenerator;
  assets: AssetManager;
  registry: RendererRegistry;
  setScenes: (scenes: AnimatedScene[], fps?: number) => void;
} {
  const { width = 1920, height = 1080, fps = 30, scenes = [] } = config;

  const registry = new RendererRegistry();
  const assets = new AssetManager();

  // Register all renderers
  const { TextRenderer } = require('./renderers/text-renderer.js');
  const { ImageRenderer } = require('./renderers/image-renderer.js');
  const { ShapeRenderer } = require('./renderers/shape-renderer.js');

  registry.register(new TextRenderer());
  registry.register(new ImageRenderer());
  registry.register(new ShapeRenderer());

  let generator = new AnimatedFrameGenerator(
    { width, height, fps, scenes },
    registry,
    assets
  );

  return {
    generator,
    assets,
    registry,
    setScenes: (newScenes: AnimatedScene[], newFps?: number) => {
      generator = new AnimatedFrameGenerator(
        { width, height, fps: newFps ?? fps, scenes: newScenes },
        registry,
        assets
      );
    },
  };
}

import { describe, it, expect, beforeEach } from 'vitest';
import { AnimatedFrameGenerator, createAnimatedRenderer } from '../../src/render/animated-frame-generator.js';
import { RendererRegistry } from '../../src/render/renderer-registry.js';
import { AssetManager } from '../../src/render/asset-manager.js';
import { TextRenderer } from '../../src/render/renderers/text-renderer.js';
import { ShapeRenderer } from '../../src/render/renderers/shape-renderer.js';

describe('AnimatedFrameGenerator', () => {
  let registry: RendererRegistry;
  let assets: AssetManager;

  beforeEach(() => {
    registry = new RendererRegistry();
    registry.register(new TextRenderer());
    registry.register(new ShapeRenderer());
    assets = new AssetManager();
  });

  describe('Success Criteria 1: Element A to B with custom easing', () => {
    it('animates element position with bounce easing', () => {
      const generator = new AnimatedFrameGenerator(
        {
          width: 100,
          height: 100,
          fps: 30,
          scenes: [{
            duration: 1,  // 30 frames
            background: '#000000',
            elements: [{
              type: 'shape',
              shape: 'circle',
              x: 10,
              y: 50,
              radius: 5,
              fill: '#ff0000',
              animations: [{
                property: 'x',
                keyframes: [
                  { frame: 0, value: 10 },
                  { frame: 30, value: 90, easing: 'easeOutBounce' },
                ],
              }],
            }],
          }],
        },
        registry,
        assets
      );

      // Get frames at 0%, 50%, 100%
      const frame0 = generator.generateFrame(0);
      const frame15 = generator.generateFrame(15);
      const frame29 = generator.generateFrame(29);

      // All frames should generate successfully
      expect(frame0.length).toBe(100 * 100 * 4);
      expect(frame15.length).toBe(100 * 100 * 4);
      expect(frame29.length).toBe(100 * 100 * 4);
    });
  });

  describe('Success Criteria 2: Enter/exit presets with timing', () => {

              type: 'shape',

              enter: {
                type: 'fade',
                duration: 0.5,  // 15 frames
              },
            }],
          }],
        },
        registry,
        assets
      );

      const totalFrames = generator.getTotalFrames();
      expect(totalFrames).toBe(60);

      // Frame 0: Should be starting fade (low opacity)
      const frame0 = generator.generateFrame(0);
      expect(frame0.length).toBe(100 * 100 * 4);
    });

    it('slides element with slide preset', () => {
      const generator = new AnimatedFrameGenerator(
        {
          width: 100,
          height: 100,
          fps: 30,
          scenes: [{
            duration: 1,
            background: '#000000',
            elements: [{
              type: 'shape',
              shape: 'rectangle',
              x: 50,
              y: 50,
              width: 20,
              height: 20,

                type: 'slide',
                duration: 0.5,

      const generator = new AnimatedFrameGenerator(
        {
          width: 100,
