  <behavior>
JobStore manages in-memory job storage with TTL cleanup:
- create(job): stores job, returns void
- get(id): returns Job | undefined
- update(id, updates): merges updates into existing job
- delete(id): removes job
- cleanup(): removes jobs older than TTL (default 24h)
- size: getter for job count

Job transitions: queued -> processing -> completed|failed

Test cases:
- create() -> get() returns same job
- update() merges partial updates
- delete() -> get() returns undefined
- cleanup() removes expired jobs, keeps fresh ones
- get() for non-existent ID returns undefined
  </behavior>
  <implementation>
Use Map<string, Job> for storage. Store created_at timestamp on each job.
cleanup() iterates, removes where Date.now() - created_at > TTL.
TTL configurable via constructor (default 24 * 60 * 60 * 1000 ms).
  </implementation>
</feature>

```typescript
import type { Job } from '../types.js';

const DEFAULT_TTL = 24 * 60 * 60 * 1000; // 24 hours

export class JobStore {
  private jobs = new Map<string, Job>();
  private ttl: number;

  constructor(ttl = DEFAULT_TTL) {
    this.ttl = ttl;
  }

  create(job: Job): void {
    this.jobs.set(job.id, job);
  }

  get(id: string): Job | undefined {
    return this.jobs.get(id);
  }

  update(id: string, updates: Partial<Job>): void {
    const existing = this.jobs.get(id);
    if (existing) {
      this.jobs.set(id, { ...existing, ...updates });
    }
  }

  delete(id: string): void {
    this.jobs.delete(id);
  }

  cleanup(): void {
    const now = Date.now();
    for (const [id, job] of this.jobs) {
      if (now - job.created_at > this.ttl) {
        this.jobs.delete(id);
      }
    }
  }

  get size(): number {
    return this.jobs.size;
  }
}
```

```typescript
import { EventEmitter } from 'node:events';
import PQueue from 'p-queue';
import { renderVideo, type RenderResult } from '../../encoder/video-renderer.js';
import { JobStore } from './job-store.js';
import type { Job } from '../types.js';
import type { VideoSpec } from '../../types/index.js';

export interface RenderJob {
  id: string;
  spec: VideoSpec;
  webhookUrl?: string;
  userId: string;
}

export interface JobQueueOptions {
  concurrency?: number;
  timeout?: number;
}

export class JobQueueService extends EventEmitter {
  private queue: PQueue;
  private store: JobStore;

  constructor(store: JobStore, options: JobQueueOptions = {}) {
    super();
    this.store = store;
    this.queue = new PQueue({
      concurrency: options.concurrency ?? 2,
      timeout: options.timeout ?? 5 * 60 * 1000, // 5 min default
    });
  }

  async enqueue(renderJob: RenderJob): Promise<void> {
    const job: Job = {
      id: renderJob.id,
      status: 'queued',
      created_at: Date.now(),
      webhook_url: renderJob.webhookUrl,
    };
    this.store.create(job);

    // Add to queue (fire and forget, status tracked in store)
    this.queue.add(async () => {
      this.store.update(renderJob.id, { status: 'processing' });
      this.emit('job:processing', renderJob.id);

      try {
        const outputPath = `outputs/${renderJob.id}.mp4`;
        const result = await renderVideo({
          scenes: renderJob.spec.scenes,
          width: renderJob.spec.output.width,
          height: renderJob.spec.output.height,
          fps: renderJob.spec.output.fps,
          outputPath,
          // audio: renderJob.spec.audio, // Add when audio in spec
        });

        this.store.update(renderJob.id, {
          status: 'completed',
          result: { download_url: `/download/${renderJob.id}.mp4` },
          completed_at: Date.now(),
        });
        this.emit('job:completed', renderJob.id, result);
      } catch (error) {
        const errorMessage = error instanceof Error ? error.message : 'Unknown error';
        this.store.update(renderJob.id, {
          status: 'failed',
          error: errorMessage,
          completed_at: Date.now(),
        });
        this.emit('job:failed', renderJob.id, error);
      }
    });
  }

  getJob(id: string): Job | undefined {
    return this.store.get(id);
  }

  get size(): number {
    return this.queue.size;
  }

  get pending(): number {
    return this.queue.pending;
  }

  onIdle(): Promise<void> {
    return this.queue.onIdle();
  }
}
```
